$('startInterval').addEvent('click', function (e) {
	// This adds 1% to the progress, every 250ms
	window.setInterval(function () {
		$('interval').set('data-progress', parseInt($('interval').get('data-progress'), 10) + 1 + '%');
	}, 250);
});
